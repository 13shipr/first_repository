# Основные команды системы

git --help  //справка
git init    //создание локального репозитория 
git status  //показать рабочий статус 
git add     //добавить файл
git log     //показать системный журнал 
git branch  //показать ветки
git checkout //перключиться на ветку
git diff     //показать изменения между комитами
git merge    //слияние веток
